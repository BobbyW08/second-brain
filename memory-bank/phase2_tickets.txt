useTasksByPriority(userId: string)
Returns all tasks for the user where status = 'active', ordered by position.
Query key: ['tasks', userId]

ts
export function useTasksByPriority(userId: string) {
  return useQuery({
    queryKey: ['tasks', userId],
    queryFn: async () => {
      const { data } = await supabase
        .from('tasks')
        .select('*')
        .eq('user_id', userId)
        .eq('status', 'active')
        .order('position')
        .throwOnError()
      return data
    },
  })
}

useCompleteTask()
Updates status to 'completed_today' and sets completed_at to now().
Input: taskId (string)
On success: invalidate ['tasks', userId]

Full optimistic update pattern (use this exact shape for all mutations):

ts
onMutate: async (taskId) => {
  await queryClient.cancelQueries({ queryKey: ['tasks', userId] })
  const previous = queryClient.getQueryData(['tasks', userId])
  queryClient.setQueryData(['tasks', userId], (old: Task[]) =>
    old.map(t => t.id === taskId
      ? { ...t, status: 'completed_today', completed_at: new Date().toISOString() }
      : t
    )
  )
  return { previous }
},
onError: (_err, _taskId, context) => {
  queryClient.setQueryData(['tasks', userId], context?.previous)
},
onSettled: () => queryClient.invalidateQueries({ queryKey: ['tasks', userId] }),

2. BUILD THE TASK ZUSTAND STORE
File path: src/stores/taskStore.ts

This store holds UI-only state that does not belong in TanStack Query.

ts
import { create } from 'zustand'

interface TaskStore {
  selectedTaskId: string | null
  setSelectedTaskId: (id: string | null) => void
  dragState: {
    isDragging: boolean
    taskId: string | null
  }
  setDragState: (state: { isDragging: boolean; taskId: string | null }) => void
}

export const useTaskStore = create<TaskStore>((set) => ({
  selectedTaskId: null,
  setSelectedTaskId: (id) => set({ selectedTaskId: id }),
  dragState: { isDragging: false, taskId: null },
  setDragState: (dragState) => set({ dragState }),
}))
